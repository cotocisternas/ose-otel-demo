**Step 6 — verify metric in Mimir (SC #3 part 2):**
```bash
# Try a few candidate metric names — exact name varies by SDK semantic conventions
for metric in 'orders_created_total' 'orders_created' 'http_server_request_duration_seconds_count' 'traces_spanmetrics_calls_total'; do
  RESP=$(curl -sf "http://localhost:9009/prometheus/api/v1/query?query=${metric}" 2>&1)
  RESULT_COUNT=$(echo "$RESP" | jq -r '.data.result | length' 2>/dev/null || echo "0")
  echo "Mimir query: $metric → $RESULT_COUNT series"
  if [ "$RESULT_COUNT" -gt 0 ]; then
    echo "$RESP" > /tmp/phase10-smoke-mimir-query.json
    METRIC_FOUND="$metric"
    break
  fi
done
test -n "${METRIC_FOUND:-}" || { echo 'FAIL: zero metrics found in Mimir for any candidate name'; exit 1; }
echo "Metric found: $METRIC_FOUND"
```
