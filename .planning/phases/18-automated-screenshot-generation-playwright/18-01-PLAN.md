### Section 3: URL builders

```javascript
function buildTempoUrl({ from, to, serviceName }) {
  const query = serviceName
    ? [{
        refId: 'A',
        queryType: 'traceqlSearch',
        filters: [{ id: 'service-name', tag: 'resource.service.name', operator: '=', value: [serviceName], type: 'tag' }],
      }]
    : [{ refId: 'A', queryType: 'traceqlSearch' }];
  const panes = { tempo: { datasource: 'tempo', queries: query, range: { from: String(from), to: String(to) } } };
  return `${GRAFANA_URL}/explore?orgId=1&schemaVersion=1&panes=${encodeURIComponent(JSON.stringify(panes))}`;
}

function buildLokiUrl({ from, to }) {
  const panes = { loki: { datasource: 'loki', queries: [{ refId: 'A', queryType: 'range', expr: '{service_name=~"order-.*"}' }], range: { from: String(from), to: String(to) } } };
  return `${GRAFANA_URL}/explore?orgId=1&schemaVersion=1&panes=${encodeURIComponent(JSON.stringify(panes))}`;
}

function buildDashboardUrl({ from, to, panelId }) {
  return `${GRAFANA_URL}/d/${DASHBOARD_UID}/?from=${from}&to=${to}&kiosk=tv${panelId ? `&viewPanel=${panelId}` : ''}`;
}
```

### Section 5: Per-capture Playwright flow

```javascript
async function captureView(browser, capture) {
  const outPath = resolve(OUTPUT_DIR, capture.output);
  if (!FORCE && existsSync(outPath)) {
    return { output: capture.output, status: 'skipped', optional: !!capture.optional };
  }

  const context = await browser.newContext();
  const page = await context.newPage();
  const { from, to } = fixedTimeRange();

  try {
    if (capture.kind === 'dashboard') {
      const url = buildDashboardUrl({ from, to, panelId: capture.panelId });
      await page.goto(url, { waitUntil: 'networkidle', timeout: 30_000 });
      await page.waitForSelector(capture.waitSelector, { state: 'visible', timeout: capture.waitTimeout ?? 15_000 });
      await page.waitForTimeout(500);
      await page.screenshot({ path: outPath, fullPage: false });
    } else if (capture.kind === 'tempo') {
      const url = buildTempoUrl({ from, to, serviceName: capture.serviceName });
      await page.goto(url, { waitUntil: 'networkidle', timeout: 30_000 });
      await page.waitForSelector(capture.waitSelector, { state: 'visible', timeout: capture.waitTimeout ?? 15_000 });
      await page.waitForTimeout(500);
      await page.screenshot({ path: outPath, fullPage: true });
    } else if (capture.kind === 'loki') {
      const url = buildLokiUrl({ from, to });
      await page.goto(url, { waitUntil: 'networkidle', timeout: 30_000 });
      await page.waitForSelector(capture.waitSelector, { state: 'visible', timeout: capture.waitTimeout ?? 15_000 });
      await page.waitForTimeout(500);
      await page.screenshot({ path: outPath, fullPage: true });
    } else if (capture.kind === 'terminal') {
      // Terminal captures run the command and render stdout as HTML — no waitSelector
      const stdout = execSync(capture.command, { cwd: REPO_ROOT, encoding: 'utf-8', maxBuffer: 50 * 1024 * 1024 });
      const html = `<!doctype html><meta charset="utf-8"><style>body{background:#1e1e1e;color:#d4d4d4;font-family:'Cascadia Code',Menlo,monospace;font-size:13px;line-height:1.4;padding:20px;white-space:pre-wrap;}</style><body>${stdout.replace(/[<>&]/g, c => ({'<':'&lt;','>':'&gt;','&':'&amp;'}[c]))}`;
      await page.setContent(html);
      await page.screenshot({ path: outPath, fullPage: true });
    } else {
      throw new Error(`Unknown capture kind: ${capture.kind}`);
    }
    console.log(`  captured: ${capture.output}`);
    return { output: capture.output, status: 'captured', optional: !!capture.optional };
  } catch (err) {
    console.warn(`  failed: ${capture.output} — ${err.message}`);
    return { output: capture.output, status: 'failed', optional: !!capture.optional };
  } finally {
    await context.close();
  }
}
```

OUTSIDE the finally block on the normal-exit path. This is safe because `copyFileSync`
is synchronous and executes even when `process.exit(1)` fires from the SIGINT handler.

```javascript
function disableTailSampling(configPath) {
  let yaml = readFileSync(configPath, 'utf-8');
  // 1. Comment out the tail_sampling processor definition block
  yaml = yaml.replace(
    /^(  tail_sampling:[\s\S]*?)^(  transform\/copy_recordpolicy:)/m,
    (match, block, next) => {
      const commented = block.split('\n').map(l => l ? '# ' + l : l).join('\n');
      return commented + next;
    }
  );
  // 2. Remove tail_sampling from the traces pipeline processors list
  yaml = yaml.replace(
    /processors: \[memory_limiter, tail_sampling, transform\/copy_recordpolicy, batch\]/,
    'processors: [memory_limiter, transform/copy_recordpolicy, batch]'
  );
  writeFileSync(configPath, yaml, 'utf-8');
}

// Synchronous restore — safe to call from process.on('exit') handler and finally blocks.
function restoreConfigSync() {
  try {
    if (existsSync(OTELCOL_CONFIG_BAK)) {
      copyFileSync(OTELCOL_CONFIG_BAK, OTELCOL_CONFIG);
    }
  } catch (err) {
    process.stderr.write(`ERROR: failed to restore otelcol-config.yaml: ${err.message}\n`);
    process.stderr.write('Manual fix: cp infra/observability/otelcol-config.yaml.bak infra/observability/otelcol-config.yaml\n');
  }
}

async function waitForCollectorHealthy(timeoutMs = 30_000) {
  // P18-2: poll :13133/ until 200 OK or timeout — must be after docker compose restart returns
  const deadline = Date.now() + timeoutMs;
  while (Date.now() < deadline) {
    try {
      const resp = await fetch('http://localhost:13133/');
      if (resp.ok) return;
    } catch { /* not ready yet */ }
    await sleep(1_000);
  }
  throw new Error('otel-collector health check timed out after 30s on :13133/');
}

function restartCollector() {
  // docker compose restart uses SERVICE name 'otel-collector', not container_name 'ose-otel-collector'
  execSync('docker compose restart otel-collector', { cwd: REPO_ROOT, stdio: 'inherit' });
}

// withTailSamplingDisabled captures the OFF frame inside fn().
// After it returns (no throw), the caller runs restartCollector() + warmup for the ON frame.
async function withTailSamplingDisabled(fn) {
  copyFileSync(OTELCOL_CONFIG, OTELCOL_CONFIG_BAK);
  try {
    disableTailSampling(OTELCOL_CONFIG);
    restartCollector();
    await waitForCollectorHealthy();
    await sleep(WARMUP_MS);  // D-T2: 30s warm-up after health-check passes
    await fn();
  } finally {
    // P18-3: synchronous config restore — runs even on throw or process.exit(1).
    // The process.on('exit') handler below also calls restoreConfigSync() as a second safety net.
    restoreConfigSync();
    // NOTE: async restartCollector() + warmup for the ON frame intentionally NOT here.
    // They are on the normal-exit path after withTailSamplingDisabled() returns.
    // If fn() threw or SIGINT fired, the caller handles the error; the config is restored on disk.
  }
  // Reached only on normal exit (no throw from fn() or withTailSamplingDisabled).
  // Restart + warm-up for the ON-frame capture.
  restartCollector();
  await waitForCollectorHealthy();
  await sleep(WARMUP_MS);  // D-T2: 30s warm-up before ON-frame capture
}
```

```javascript
async function captureTagCyclingEntries(browser, tagEntries) {
  // Group by tag so each tag spins up infra once
  const byTag = {};
  for (const c of tagEntries) {
    (byTag[c.tag] ||= []).push(c);
  }
  const results = [];
  for (const [tag, captures] of Object.entries(byTag)) {
    // Skip if ALL outputs for this tag already exist (D-I1)
    if (!FORCE && captures.every(c => existsSync(resolve(OUTPUT_DIR, c.output)))) {
      for (const c of captures) results.push({ output: c.output, status: 'skipped', optional: !!c.optional });
      continue;
    }
    console.log(`\n=== tag: ${tag} ===`);
    cleanupWorktree();
    execSync(`git worktree add "${WORKTREE_DIR}" ${tag}`, { cwd: REPO_ROOT, stdio: 'inherit' });
    let devPid = null;
    let tagLoadProc = null;
    try {
      // Build at tag
      execSync('mvn -T 1C -DskipTests -q clean install', { cwd: WORKTREE_DIR, stdio: 'inherit' });
      // Start dev (background)
      devPid = spawn('mise', ['run', 'dev'], {
        cwd: WORKTREE_DIR,
        env: { ...process.env },
        stdio: 'ignore',
        detached: false,
      });
      await sleep(25_000);  // Spring Boot startup
      // Start load (skip for step-01-baseline — no apps ready yet)
      if (tag !== 'step-01-baseline') {
        tagLoadProc = spawn(resolve(REPO_ROOT, 'scripts', 'load.sh'), [], {
          cwd: REPO_ROOT,
          env: { ...process.env },
          stdio: 'ignore',
          detached: false,
        });
        await sleep(WARMUP_MS);
      }
      // Capture
      for (const c of captures) {
        const r = await captureView(browser, c);
        results.push(r);
      }
    } finally {
      if (tagLoadProc && !tagLoadProc.killed) try { tagLoadProc.kill('SIGTERM'); } catch { /* ignore */ }
      if (devPid && !devPid.killed) try { devPid.kill('SIGTERM'); } catch { /* ignore */ }
      try { execSync(`pkill -f "${MVN_KILL_PATTERN}"`, { stdio: 'ignore' }); } catch { /* ignore */ }
      await sleep(3_000);
      cleanupWorktree();
    }
  }
  return results;
}
```
